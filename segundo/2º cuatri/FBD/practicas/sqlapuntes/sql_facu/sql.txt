¿Qué es sql?
	Structured Query Language
	Nos permite trabajar con bases de datos relacionales.
	
Crear base de datos:
	CREATE DATABASE "USUARIO"
	
Tipos de datos:
	INTEGER -> int
	TEXT -> texto
	BLOB -> binario
	REAL -> float, valores con coma
	NUMERIC -> números con precisión numérica muy alta, por ejemplo pi, no tiene limite por lo q es más lento
	
Crear tabla:
	CREATE TABLE "usuarios" (
		"nombre"	TEXT,
		"DNI"	INTEGER,
		"edad"	INTEGER DEFAULT 15	//la edad por defecto es 15
	);
	
	En las tablas un campo equivale a una columna
	Un registro es una fila, contiene la información de cada uno de los campos
	El valor del campo es lo que contiene un apartado, un determinado registro con un determinado campo, por ej casilla 1,1

GROUP BY (agrupar por un atributo y devuelve el primer valor de cada uno, por ej si le pedimos el nombre nos va a devolver el nombre del primero)
	CON EL WHERE NO PODEMOS HACER REFERENCIA A UNA FUNCIÓN DE AGRUPACIÓN, POR LO Q TENEMOS QUE USAR HAVING Q A DIFERENCIA DE WHERE ESTE FILTRA POR GRUPOS EN VEZ DE POR REGISTROS
	
		SELECT SupplierID, ROUND(AVG(Price)) AS promedioID FROM Products
		GROUP BY SupplierID
		ORDER BY promedioID DESC 	//va a realizar el promedio del precio de cada supplierID
		
		SELECT SupplierID, ROUND(AVG(Price)) AS promedioID FROM Products
		WHERE ProductName IS NOT NULL
		GROUP BY SupplierID
		HAVING promedioID > 40		//podemos usar clausulas como between HAVING promedioID BETWEEN 40 AND 150
		ORDER BY promedioID DESC	//devolvera los que tengan un promedio mayor a 40
		
		SELECT ProductID, SUM(Quantity) AS cantidadTotal FROM OrderDetails
		GROUP BY ProductID
		ORDER BY cantidadTotal DESC //suma las cantidades y las divide por productid
		LIMIT 1 //nos devolverá el producto más vendido
		
SUBCONSULTAS(consulta dentro dde otra consulta)
	SELECT ProductID, 
		Quantity,
		(SELECT ProductName FROM Products WHERE OrderDetails.ProductID = ProductID) AS nombre
	FROM OrderDetails //va a mostrar una tabla con producid, con su cantidad y su nombre
	
	SELECT ProductID, SUM(Quantity) as total_vendido,
		(SELECT Price FROM Products where OD.ProductID = ProductID) as precio,
		SUM((Quantity)) * (SELECT Price FROM Products where OD.ProductID = ProductID) as total_recaudado
	FROM [OrderDetails] OD
	GROUP by ProductID	//crea una tabla con el total vendido
	HAVING total_recaudado < 100 // nos devuelve los que hayan recaudado menos de 100
	
	Ordenar por un atributo de la tabla sin mostrarlo
		SELECT ProductID, SUM(Quantity) as total_vendido,
			(SELECT ProductName FROM Products where OD.ProductID = ProductID) as name,
			SUM((Quantity)) * (SELECT Price FROM Products where OD.ProductID = ProductID) as total_recaudado
		FROM [OrderDetails] OD
		WHERE (SELECT Price FROM Products where OD.ProductID = ProductID) > 40
		GROUP by ProductID	
	
	SELECT SUM(Quantity*(SELECT Price FROM Products where OrderDetails.ProductID = ProductID)) from OrderDetails //nos da la suma del total vendido

NORMALIZACIÓN	//asegurar la eficiencia de la base de datos
	forma normal 1	//consiste en garantizar que cada atributo en una tabla tenga un valor único atómico, no debería haber atributos repetidos en una fila para una clave primaria, por ej, tener una columna con dni, otra con nombre y otra con apellido
		
	forma normal 2	//cada atributo que no sea una clave debe depender de la clave primaria, por ej en una tabla PedidoID, ClienteID, Fecha del pedido, ProductoID, Nombre del producto, Precio del producto, Cantidad del producto, no cumple la segunda forma normal ya que Nombre del producto y Precio del producto no dependen al 100% de PedidoID, sino que dependen también de ProductoID, lo que hace que deban estar en otra tabla
	
	forma normal 3	// cada atributo debe depender directamente de la clave primaria y no de atributos que no sean claves primarias, dependencia transitiva, que dependa del atributo que dependa del principal
